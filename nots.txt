0.gti init                                                      'برای درست کردن فایل .git'
1.git status                                                    'برای گرفتن آمار که چه چیز هایی تغییر کرده است'
2.git log                                                       'برای فهمیدن سیو ها و پیز های جدید '
3.git add . or git add <file>                                   'وارد کردن فایل ها به استیج'
4.add commit -m "text"                                          'برای کامیت کردن آنها به استیج'
5.git branch                                                    'ساخت یک برنچ جدید'
6.git branch -a                                                 'دیدن همه برنچ های ساخته شده'
7.git checkout <name of branch>                                 'وارد شدن به یک برنچ دیگه'
8.git checkout -- . or git checkout -- <file>                   'برگشتن به حالت قبلی قبل از ادد دات'
9.git reset HEAD                                                'خارج کردن از استیج'
10.git merge <file>                                             'متصل کردن یک برنچ به برنچ دیگه'
